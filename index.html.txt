        function setupClimbModal() {
            const rows = document.querySelectorAll('#climbsTableBody tr');
            const overlay = document.getElementById('climbModalOverlay');
            const closeBtn = document.getElementById('modalCloseBtn');
            const modalTitle = document.getElementById('modalTitle');
            const modalMeta = document.getElementById('modalMeta');
            const modalSummary = document.getElementById('modalSummary');
            const modalLink = document.getElementById('modalLink');

            function openModal(id) {
                const climb = climbDetails[id];
                if (!climb) return;

                modalTitle.textContent = climb.name;
                modalMeta.innerHTML =
                    `<span><strong>Distance:</strong> ${climb.distance}</span>` +
                    `<span><strong>Elevation:</strong> ${climb.elevation}</span>` +
                    `<span><strong>Avg Gradient:</strong> ${climb.gradient}</span>` +
                    `<span><strong>Difficulty:</strong> ${climb.difficulty}</span>`;
                modalSummary.textContent = climb.summary;
                modalLink.innerHTML =
                    `<a href="${climb.link}" target="_blank" rel="noopener noreferrer">Open full climb guide</a>`;

                overlay.classList.add('active');
                overlay.setAttribute('aria-hidden', 'false');
            }

            function closeModal() {
                overlay.classList.remove('active');
                overlay.setAttribute('aria-hidden', 'true');
            }

            rows.forEach(row => {
                row.addEventListener('click', () => {
                    const id = row.getAttribute('data-climb-id');
                    openModal(id);
                });
            });

            closeBtn.addEventListener('click', closeModal);
            overlay.addEventListener('click', (e) => {
                if (e.target === overlay) closeModal();
            });
            window.addEventListener('keydown', (e) => {
                if (e.key === 'Escape') closeModal();
            });
        }

        window.addEventListener('load', () => {
            drawWeatherChart();
            setupClimbModal();
        });
        window.addEventListener('resize', drawWeatherChart);
    </script>
